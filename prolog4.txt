%99. Определите предикат subsum(+Set,+Sum,?SubSet)
% такой, что Set является списком чисел, 
% SubSet - подмножеством этих чисел, а сумма чисел в SubSet рав-на Sum, например:
% ?- subsum([1,2,5,3,2],5,Sub).
% Sub = [1,2,2];
% Sub=[2,3];
% Sub=[5];
% ...

p([H|T],N,[H|T1]):- H =< N , N1 is N-H, p(T,N1,T1).
p([_|T],N,T1):- p(T,N,T1).
p([],0,[]).
